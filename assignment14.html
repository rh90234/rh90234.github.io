SELECT (COUNT(*)) FROM Customers;
SELECT Country, (COUNT(*)) FROM Customers GROUP BY Country;
SELECT Country, (COUNT(*)) FROM Customers GROUP BY Country ORDER BY (COUNT(*)) DESC, Country;
SELECT Country, (COUNT(*)) FROM Customers GROUP BY Country HAVING (COUNT(*)) > 10 ORDER BY (COUNT(*)) DESC, Country;
